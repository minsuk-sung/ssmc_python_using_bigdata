    client_id = "ppihLuN2pr20zbkL0CAJ"

    encText = urllib.parse.quote(s)
    url = "https://openapi.naver.com/v1/map/geocode?query=" + encText # json °á°ú

        aa= response_body.decode('utf-8') 
        bb = json.loads(aa)
